ما هي PHP؟

PHP (PHP: Hypertext Preprocessor) هي لغة برمجة مفتوحة المصدر تُستخدم بشكل أساسي لتطوير تطبيقات الويب. تتميز PHP بسهولة التعلم والاستخدام، وتدعم العديد من قواعد البيانات.

مميزات PHP:

  • لغة مفتوحة المصدر ومجانية
  • سهلة التعلم للمبتدئين
  • تدعم العديد من قواعد البيانات
  • متوافقة مع جميع أنظمة التشغيل
  • مجتمع كبير من المطورين

أول برنامج PHP

<?php
// هذا تعليق في PHP
echo "مرحباً بالعالم!";
echo "<br>";
echo "أهلاً وسهلاً في دورة PHP";
?>

شرح الكود:

  • <?php - بداية كود PHP
  • // - تعليق من سطر واحد
  • echo - أمر طباعة النص
  • ?> - نهاية كود PHP

المتغيرات

المتغير في PHP هو حاوية لتخزين البيانات. تبدأ جميع المتغيرات في PHP بعلامة الدولار ($).

أمثلة على المتغيرات

<?php
// متغير نصي
$name = "أحمد";
$message = "مرحباً";

// متغير رقمي
$age = 25;
$price = 99.99;

// متغير منطقي
$is_student = true;
$is_working = false;

// طباعة المتغيرات
echo $name;
echo "<br>";
echo "العمر: " . $age;
echo "<br>";
echo "السعر: $" . $price;
?>

شرح الكود:

  • $name - متغير نصي (String)
  • $age - متغير رقمي صحيح (Integer)
  • $price - متغير رقمي عشري (Float)
  • $is_student - متغير منطقي (Boolean)
  • . - عامل ربط النصوص

أنواع البيانات

أنواع البيانات الأساسية

  • String (نصي): لتخزين النصوص
  • Integer (صحيح): لتخزين الأرقام الصحيحة
  • Float (عشري): لتخزين الأرقام العشرية
  • Boolean (منطقي): true أو false
  • Array (مصفوفة): لتخزين عدة قيم

أمثلة على أنواع البيانات

<?php
// String
$text = "مرحباً بالعالم";
$number_as_text = "123";

// Integer
$whole_number = 42;
$negative_number = -10;

// Float
$decimal = 3.14;
$price = 19.99;

// Boolean
$is_true = true;
$is_false = false;

// Array
$colors = array("أحمر", "أخضر", "أزرق");
$numbers = [1, 2, 3, 4, 5];

// طباعة نوع البيانات
echo gettype($text); // string
echo "<br>";
echo gettype($whole_number); // integer
?>

المصفوفات (Arrays)

المصفوفات

المصفوفة هي متغير يمكنه تخزين عدة قيم في متغير واحد. يمكن الوصول للعناصر باستخدام الفهرس.

أمثلة على المصفوفات

<?php
// مصفوفة بسيطة
$fruits = array("تفاح", "موز", "برتقال");
// أو
$fruits = ["تفاح", "موز", "برتقال"];

// الوصول للعناصر
echo $fruits[0]; // تفاح
echo "<br>";
echo $fruits[1]; // موز

// مصفوفة بروابط نصية
$person = array(
    "name" => "أحمد",
    "age" => 25,
    "city" => "دمشق"
);

echo $person["name"]; // أحمد
echo "<br>";
echo $person["age"]; // 25

// إضافة عنصر جديد
$fruits[] = "عنب";
$person["job"] = "مبرمج";
?>

هياكل التحكم

الشروط (If Statements)

أهداف التعلم

  • تعلم استخدام if, else, elseif
  • فهم عوامل المقارنة
  • تعلم العوامل المنطقية

أمثلة على الشروط

<?php
$age = 20;
$grade = 85;

// if بسيط
if ($age >= 18) {
    echo "أنت بالغ";
}

// if else
if ($grade >= 60) {
    echo "نجحت في الامتحان";
} else {
    echo "راسب في الامتحان";
}

// if elseif else
if ($grade >= 90) {
    echo "ممتاز";
} elseif ($grade >= 80) {
    echo "جيد جداً";
} elseif ($grade >= 70) {
    echo "جيد";
} else {
    echo "مقبول";
}

// عوامل المقارنة
$num1 = 10;
$num2 = 20;

if ($num1 == $num2) {  // يساوي
    echo "الرقمان متساويان";
}

if ($num1 != $num2) {  // لا يساوي
    echo "الرقمان غير متساويين";
}

if ($num1 < $num2) {   // أصغر من
    echo "الرقم الأول أصغر";
}

// العوامل المنطقية
$is_student = true;
$has_id = true;

if ($is_student && $has_id) {  // و
    echo "يمكنك الدخول";
}

if ($is_student || $has_id) {  // أو
    echo "يمكنك الدخول";
}
?>

الحلقات (Loops)

أنواع الحلقات

  • for: للحلقات المحددة العدد
  • while: للحلقات غير المحددة
  • foreach: للتكرار عبر المصفوفات

أمثلة على الحلقات

<?php
// حلقة for
echo "الأرقام من 1 إلى 5:<br>";
for ($i = 1; $i <= 5; $i++) {
    echo $i . "<br>";
}

// حلقة while
$count = 1;
echo "<br>الأرقام الزوجية:<br>";
while ($count <= 10) {
    if ($count % 2 == 0) {
        echo $count . "<br>";
    }
    $count++;
}

// حلقة foreach للمصفوفات
$colors = ["أحمر", "أخضر", "أزرق", "أصفر"];
echo "<br>الألوان:<br>";
foreach ($colors as $color) {
    echo "- " . $color . "<br>";
}

// foreach للمصفوفات المرتبطة
$person = [
    "name" => "أحمد",
    "age" => 25,
    "city" => "دمشق"
];

echo "<br>معلومات الشخص:<br>";
foreach ($person as $key => $value) {
    echo $key . ": " . $value . "<br>";
}
?>

الدوال (Functions)

إنشاء واستخدام الدوال

أهداف التعلم

  • تعلم كيفية إنشاء الدوال
  • فهم المعاملات والقيم المرجعة
  • تعلم الدوال المدمجة في PHP

إنشاء الدوال

<?php
// دالة بسيطة
function sayHello() {
    echo "مرحباً بالعالم!";
}

// استدعاء الدالة
sayHello();

// دالة مع معاملات
function greet($name) {
    echo "مرحباً " . $name;
}

greet("أحمد");

// دالة مع عدة معاملات
function add($a, $b) {
    return $a + $b;
}

$result = add(5, 3);
echo "<br>النتيجة: " . $result;

// دالة مع قيمة افتراضية
function greetWithDefault($name = "زائر") {
    echo "<br>مرحباً " . $name;
}

greetWithDefault(); // ستستخدم القيمة الافتراضية
greetWithDefault("سارة");

// دالة مع نوع البيانات
function calculateArea(float $length, float $width): float {
    return $length * $width;
}

$area = calculateArea(5.5, 3.2);
echo "<br>المساحة: " . $area;
?>

الدوال المدمجة

بعض الدوال المهمة

  • strlen() - طول النص
  • strtoupper() - تحويل للحروف الكبيرة
  • strtolower() - تحويل للحروف الصغيرة
  • date() - التاريخ والوقت
  • count() - عدد عناصر المصفوفة

أمثلة على الدوال المدمجة

<?php
$text = "مرحباً بالعالم";

// دوال النصوص
echo "طول النص: " . strlen($text) . "<br>";
echo "النص بالأحرف الكبيرة: " . strtoupper($text) . "<br>";
echo "النص بالأحرف الصغيرة: " . strtolower($text) . "<br>";

// دوال التاريخ
echo "التاريخ الحالي: " . date("Y-m-d") . "<br>";
echo "الوقت الحالي: " . date("H:i:s") . "<br>";

// دوال المصفوفات
$numbers = [1, 2, 3, 4, 5];
echo "عدد العناصر: " . count($numbers) . "<br>";
echo "أكبر رقم: " . max($numbers) . "<br>";
echo "أصغر رقم: " . min($numbers) . "<br>";

// دوال الأرقام
$number = 3.14159;
echo "الرقم مقرب: " . round($number, 2) . "<br>";
echo "الرقم مرفوع: " . ceil($number) . "<br>";
echo "الرقم مقلوب: " . floor($number) . "<br>";
?>

معالجة النماذج (Forms)

أهداف التعلم

  • تعلم كيفية معالجة بيانات النماذج
  • فهم $_GET و $_POST
  • تعلم التحقق من صحة البيانات

معالجة النماذج

<?php
// التحقق من إرسال النموذج
if ($_POST) {
    $name = $_POST['name'];
    $email = $_POST['email'];
    $message = $_POST['message'];
    
    // التحقق من صحة البيانات
    if (empty($name)) {
        echo "الاسم مطلوب<br>";
    } elseif (empty($email)) {
        echo "البريد الإلكتروني مطلوب<br>";
    } elseif (!filter_var($email, FILTER_VALIDATE_EMAIL)) {
        echo "البريد الإلكتروني غير صحيح<br>";
    } else {
        echo "تم إرسال البيانات بنجاح!<br>";
        echo "الاسم: " . htmlspecialchars($name) . "<br>";
        echo "البريد: " . htmlspecialchars($email) . "<br>";
        echo "الرسالة: " . htmlspecialchars($message) . "<br>";
    }
}
?>

جلسات المستخدم (Sessions)

الجلسات

الجلسات تسمح بتخزين معلومات المستخدم عبر عدة صفحات. تبدأ الجلسة بـ session_start() وتستخدم $_SESSION لتخزين البيانات.

استخدام الجلسات

<?php
// بدء الجلسة
session_start();

// تخزين بيانات في الجلسة
$_SESSION['username'] = "أحمد";
$_SESSION['user_id'] = 123;
$_SESSION['login_time'] = date('Y-m-d H:i:s');

// عرض بيانات الجلسة
echo "اسم المستخدم: " . $_SESSION['username'] . "<br>";
echo "معرف المستخدم: " . $_SESSION['user_id'] . "<br>";
echo "وقت تسجيل الدخول: " . $_SESSION['login_time'] . "<br>";

// تحديث بيانات الجلسة
$_SESSION['last_activity'] = date('Y-m-d H:i:s');

// حذف متغير من الجلسة
unset($_SESSION['last_activity']);

// إنهاء الجلسة
// session_destroy();
?>

قواعد البيانات

الاتصال بقاعدة البيانات

أهداف التعلم

  • تعلم الاتصال بقاعدة بيانات MySQL
  • فهم PDO و MySQLi
  • تعلم تنفيذ استعلامات SQL

الاتصال بقاعدة البيانات باستخدام PDO

<?php
try {
    // إعدادات الاتصال
    $host = 'localhost';
    $dbname = 'my_database';
    $username = 'root';
    $password = '';
    
    // إنشاء الاتصال
    $pdo = new PDO("mysql:host=$host;dbname=$dbname;charset=utf8", $username, $password);
    $pdo->setAttribute(PDO::ATTR_ERRMODE, PDO::ERRMODE_EXCEPTION);
    
    echo "تم الاتصال بقاعدة البيانات بنجاح!";
    
} catch(PDOException $e) {
    echo "خطأ في الاتصال: " . $e->getMessage();
}

// إنشاء جدول
$sql = "CREATE TABLE IF NOT EXISTS users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
)";

$pdo->exec($sql);
echo "<br>تم إنشاء الجدول بنجاح!";
?>

عمليات قاعدة البيانات (CRUD)

عمليات CRUD

  • Create: إدراج بيانات جديدة
  • Read: قراءة البيانات
  • Update: تحديث البيانات
  • Delete: حذف البيانات

أمثلة على عمليات CRUD

<?php
// افتراض وجود الاتصال $pdo

// CREATE - إدراج بيانات جديدة
$stmt = $pdo->prepare("INSERT INTO users (name, email) VALUES (?, ?)");
$stmt->execute(['أحمد محمد', 'ahmed@example.com']);
echo "تم إدراج المستخدم بنجاح!<br>";

// READ - قراءة البيانات
$stmt = $pdo->query("SELECT * FROM users");
$users = $stmt->fetchAll(PDO::FETCH_ASSOC);

echo "<h3>قائمة المستخدمين:</h3>";
foreach ($users as $user) {
    echo "ID: " . $user['id'] . " - ";
    echo "الاسم: " . $user['name'] . " - ";
    echo "البريد: " . $user['email'] . "<br>";
}

// UPDATE - تحديث البيانات
$stmt = $pdo->prepare("UPDATE users SET name = ? WHERE id = ?");
$stmt->execute(['أحمد علي', 1]);
echo "<br>تم تحديث المستخدم بنجاح!<br>";

// DELETE - حذف البيانات
$stmt = $pdo->prepare("DELETE FROM users WHERE id = ?");
$stmt->execute([1]);
echo "تم حذف المستخدم بنجاح!<br>";

// البحث في البيانات
$search = 'أحمد';
$stmt = $pdo->prepare("SELECT * FROM users WHERE name LIKE ?");
$stmt->execute(["%$search%"]);
$results = $stmt->fetchAll(PDO::FETCH_ASSOC);

echo "<h3>نتائج البحث:</h3>";
foreach ($results as $user) {
    echo "الاسم: " . $user['name'] . "<br>";
}
?>
