مهام الطلاب - مشاريع تطوير الويب

مجموعة من المشاريع العملية المصممة لتحسين مهارات المبتدئين في تطوير الويب

هذه المجموعة من المهام مصممة خصيصاً للمبتدئين في تطوير الويب. كل مهمة تركز على مهارة معينة وتتضمن تعليمات واضحة وحلول كاملة. ابدأ بالمهام البسيطة وتقدم تدريجياً إلى المهام الأكثر تعقيداً.

1

ساعة رقمية

مبتدئ

أنشئ ساعة رقمية تفاعلية تعرض الوقت الحالي وتحدث نفسها كل ثانية. هذا المشروع سيساعدك على فهم كيفية التعامل مع الوقت في JavaScript وإنشاء عناصر تفاعلية.

المتطلبات:

  • HTML: إنشاء div لعرض الساعة
  • CSS: تصميم الساعة مثل شاشة رقمية حديثة
  • JavaScript: تحديث الوقت كل ثانية

الحل:

HTML
CSS
JavaScript

<!DOCTYPE html>
<html lang="ar">
<head>
    <meta charset="UTF-8">
    <title>ساعة رقمية</title>
    <link rel="stylesheet" href="style.css">
</head>
<body>
    <div class="clock-container">
        <div class="digital-clock" id="clock">
            00:00:00
        </div>
    </div>
    <script src="script.js"></script>
</body>
</html>
                            

body {
    margin: 0;
    padding: 0;
    background: linear-gradient(135deg, #667eea 0%, #764ba2 100%);
    font-family: 'Arial', sans-serif;
    display: flex;
    justify-content: center;
    align-items: center;
    min-height: 100vh;
}

.clock-container {
    background: rgba(0, 0, 0, 0.8);
    padding: 2rem;
    border-radius: 20px;
    box-shadow: 0 10px 30px rgba(0, 0, 0, 0.3);
}

.digital-clock {
    font-size: 4rem;
    color: #00ff00;
    font-family: 'Courier New', monospace;
    text-align: center;
    text-shadow: 0 0 20px #00ff00;
    letter-spacing: 0.1em;
}
                            

function updateClock() {
    const now = new Date();
    const hours = now.getHours().toString().padStart(2, '0');
    const minutes = now.getMinutes().toString().padStart(2, '0');
    const seconds = now.getSeconds().toString().padStart(2, '0');
    
    const timeString = `${hours}:${minutes}:${seconds}`;
    document.getElementById('clock').textContent = timeString;
}

// تحديث الساعة كل ثانية
setInterval(updateClock, 1000);

// تحديث الساعة فوراً عند تحميل الصفحة
updateClock();

اختبار تفاعلي

مبتدئ

أنشئ اختبار تفاعلي بأسئلة متعددة الخيارات مع إمكانية تحديد الإجابات الصحيحة والخاطئة. سيساعدك هذا المشروع على فهم كيفية التعامل مع الأحداث والتفاعل مع المستخدم.

المتطلبات:

  • HTML: عرض سؤال مع خيارات متعددة
  • CSS: تمييز الإجابات الصحيحة والخاطئة
  • JavaScript: فحص الإجابة عند النقر

HTML
CSS
JavaScript

            <button class="option" data-answer="wrong">Python</button>

document.addEventListener('DOMContentLoaded', function() {
    const options = document.querySelectorAll('.option');
    const result = document.getElementById('result');
    
    options.forEach(option => {
        option.addEventListener('click', function() {
            const isCorrect = this.dataset.answer === 'correct';
            
            // إزالة الأحداث من جميع الخيارات
            options.forEach(opt => opt.style.pointerEvents = 'none');
            
            // تمييز الإجابة المختارة
            if (isCorrect) {
                this.classList.add('correct');
                result.textContent = 'إجابة صحيحة! 🎉';
                result.style.background = '#d4edda';
                result.style.color = '#155724';
            } else {
                this.classList.add('wrong');
                result.textContent = 'إجابة خاطئة! ❌';
                result.style.background = '#f8d7da';
                result.style.color = '#721c24';
                
                // تمييز الإجابة الصحيحة
                options.forEach(opt => {
                    if (opt.dataset.answer === 'correct') {
                        opt.classList.add('correct');
                    }
                });
            }
        });
    });
});

قائمة المهام

مبتدئ

أنشئ تطبيق قائمة مهام بسيط يتيح للمستخدم إضافة وحذف المهام وتمييزها كمكتملة. هذا المشروع سيساعدك على فهم إدارة الحالة والتلاعب بـ DOM.

  • HTML: حقل إدخال + زر + قائمة المهام
  • CSS: تصميم المهام مع خط للعناصر المكتملة

.task-item.completed {
    text-decoration: line-through;
    opacity: 0.6;
}

.task-text {
    flex: 1;

    color: white;
    border: none;
    padding: 0.5rem 1rem;

document.addEventListener('DOMContentLoaded', function() {
    const taskInput = document.getElementById('taskInput');
    const addBtn = document.getElementById('addBtn');
    const taskList = document.getElementById('taskList');
    
    addBtn.addEventListener('click', addTask);
    taskInput.addEventListener('keypress', function(e) {
        if (e.key === 'Enter') {
            addTask();
        }
    });
    
    function addTask() {
        const taskText = taskInput.value.trim();
        if (taskText === '') return;
        
        const taskItem = document.createElement('li');
        taskItem.className = 'task-item';
        
        taskItem.innerHTML = `
            <span class="task-text">${taskText}</span>
            <button class="delete-btn">حذف</button>
        `;
        
        // إضافة حدث النقر لتمييز المهمة كمكتملة
        taskItem.addEventListener('click', function() {
            this.classList.toggle('completed');
        });
        
        // إضافة حدث النقر لحذف المهمة
        taskItem.querySelector('.delete-btn').addEventListener('click', function(e) {
            e.stopPropagation();
            taskItem.remove();
        });
        
        taskList.appendChild(taskItem);
        taskInput.value = '';
    }
});

معرض الصور مع النافذة المنبثقة

document.addEventListener('DOMContentLoaded', function() {
    const modal = document.getElementById('modal');
    const modalImg = document.getElementById('modalImg');
    const caption = document.getElementById('caption');
    const close = document.getElementsByClassName('close')[0];
    const galleryItems = document.querySelectorAll('.gallery-item');
    
    // فتح النافذة المنبثقة عند النقر على صورة
    galleryItems.forEach(item => {
        item.addEventListener('click', function() {
            modal.style.display = 'block';
            modalImg.src = this.src;
            caption.innerHTML = this.alt;
        });
    });
    
    // إغلاق النافذة المنبثقة عند النقر على X
    close.addEventListener('click', function() {
        modal.style.display = 'none';
    });
    
    // إغلاق النافذة المنبثقة عند النقر خارج الصورة
    modal.addEventListener('click', function(e) {
        if (e.target === modal) {
            modal.style.display = 'none';
        }
    });
    
    // إغلاق النافذة المنبثقة بمفتاح Escape
    document.addEventListener('keydown', function(e) {
        if (e.key === 'Escape') {
            modal.style.display = 'none';
        }
    });
});

تبديل الوضع المظلم/المضيء

مبتدئ

أنشئ مفتاح تبديل بين الوضع المظلم والمضيء للموقع. سيساعدك هذا المشروع على فهم كيفية إدارة الثيمات وتغيير الأنماط ديناميكياً.

المتطلبات:

  • HTML: إضافة زر التبديل
  • CSS: ثيمين (مضيء + مظلم)
  • JavaScript: تبديل الأنماط ديناميكياً

الحل:

HTML
CSS
JavaScript

<div class="theme-switcher">
    <button id="themeToggle" class="theme-btn">
        <i class="fas fa-moon"></i>
    </button>
</div>

<div class="content">
    <h1>تبديل الوضع المظلم/المضيء</h1>
    <p>انقر على الزر أعلاه لتغيير ثيم الموقع.</p>
</div>
                            

/* الثيم المضيء (الافتراضي) */
:root {
    --bg-color: #ffffff;
    --text-color: #333333;
    --card-bg: #f8f9fa;
    --border-color: #e9ecef;
}

/* الثيم المظلم */
[data-theme="dark"] {
    --bg-color: #1a1a1a;
    --text-color: #ffffff;
    --card-bg: #2d2d2d;
    --border-color: #404040;
}

body {
    background-color: var(--bg-color);
    color: var(--text-color);
    transition: all 0.3s ease;
}

.theme-switcher {
    position: fixed;
    top: 20px;
    right: 20px;
    z-index: 1000;
}

.theme-btn {
    background: var(--card-bg);
    border: 2px solid var(--border-color);
    border-radius: 50%;
    width: 50px;
    height: 50px;
    cursor: pointer;
    display: flex;
    align-items: center;
    justify-content: center;
    font-size: 1.2rem;
    color: var(--text-color);
    transition: all 0.3s ease;
}

.theme-btn:hover {
    transform: scale(1.1);
}

.content {
    max-width: 800px;
    margin: 4rem auto;
    padding: 2rem;
    background: var(--card-bg);
    border-radius: 15px;
    border: 1px solid var(--border-color);
}
                            

document.addEventListener('DOMContentLoaded', function() {
    const themeToggle = document.getElementById('themeToggle');
    const body = document.body;
    
    // تحميل الثيم المحفوظ أو استخدام الافتراضي
    const savedTheme = localStorage.getItem('theme') || 'light';
    body.setAttribute('data-theme', savedTheme);
    updateThemeIcon(savedTheme);
    
    themeToggle.addEventListener('click', function() {
        const currentTheme = body.getAttribute('data-theme');
        const newTheme = currentTheme === 'dark' ? 'light' : 'dark';
        
        body.setAttribute('data-theme', newTheme);
        localStorage.setItem('theme', newTheme);
        updateThemeIcon(newTheme);
    });
    
    function updateThemeIcon(theme) {
        const icon = themeToggle.querySelector('i');
        if (theme === 'dark') {
            icon.className = 'fas fa-sun';
        } else {
            icon.className = 'fas fa-moon';
        }
    }
});

ملاحظة: هذه أمثلة على أول 5 مهام. يمكنك إضافة المهام المتبقية (6-14) بنفس الطريقة. كل مهمة تحتوي على تعليمات واضحة وحلول كاملة مع أمثلة عملية.
